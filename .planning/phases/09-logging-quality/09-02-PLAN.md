2. Add `getExitCode(error)` helper function that determines the correct exit code:
   - If CLIError with code starting with "AUTH_" -> return ExitCode.AUTH_ERROR (3)
   - If CLIError with code like "INVALID_" or "MISSING_" or "BAD_" -> return ExitCode.USAGE_ERROR (1)
   - Otherwise -> return ExitCode.SERVER_ERROR (2)
3. Update printError to accept an optional exitCode parameter and log via stderr JSON

```typescript
import { ExitCode } from "./exit-codes.js";

/**
 * Determine the correct exit code based on error type and code.
 */
export function getExitCode(error: unknown): number {
  if (error instanceof CLIError) {
    // Auth errors
    if (error.code.startsWith("AUTH_") || error.code.includes("AUTH")) {
      return ExitCode.AUTH_ERROR;
    }
    // Usage errors (invalid input, missing args, etc.)
    if (
      error.code.startsWith("INVALID_") ||
      error.code.startsWith("MISSING_") ||
      error.code.startsWith("BAD_") ||
      error.code.includes("USAGE")
    ) {
      return ExitCode.USAGE_ERROR;
    }
  }
  // Default to server error for unknown errors
  return ExitCode.SERVER_ERROR;
}
```
