```typescript
// src/commands/list.ts
import { Command } from "commander";
import { mailboxService } from "../container.js";
import { resolveProvider } from "./utils/resolve-provider.js"; // CLI-specific

export function registerListCommand(program: Command) {
  program
    .command("list")
    .description("List emails in a folder")
    .option("--account <id>", "Account ID")
    .option("--folder <name>", "Folder name", "Inbox")
    .option("--limit <number>", "Max results", "20")
    .action(async (options) => {
      try {
        const provider = await resolveProvider(options.account);
        const limit = parseInt(options.limit, 10);
        // ... validation ...
        const result = await mailboxService.list(options.folder, limit);
        console.log(JSON.stringify(result));
      } catch (err) {
        printError(err);
        process.exit(1);
      }
    });
}
```

**container.ts structure:**
```typescript
import { GmailProvider, OutlookProvider } from "./infrastructure/index.js";
import { TokenStorageImpl } from "./infrastructure/token-storage.js";
import { ConfigImpl } from "./infrastructure/config.js";
import { MailboxService, EmailService, ComposeService, AccountService } from "./services/index.js";

// Token storage and config are singletons (stateless)
const tokenStorage = new TokenStorageImpl();
const config = new ConfigImpl();

// Auth clients need initialization (OAuth2 setup)
import { initOAuthClient } from "./infrastructure/auth/oauth.js";
import { initOutlookClient } from "./infrastructure/auth/outlook-oauth.js";

// Provider factory based on account
function createProvider(account: string) {
  if (account.endsWith(":gmail")) {
    return new GmailProvider(account);
  } else if (account.endsWith(":outlook")) {
    return new OutlookProvider(account);
  }
  return new GmailProvider(account);
}

// Services receive concrete implementations
const mailboxService = new MailboxService(/* provider injected per-command via factory */);
const emailService = new EmailService(/* provider injected per-command via factory */);
const composeService = new ComposeService(/* provider injected per-command via factory */);
const accountService = new AccountService(tokenStorage, config);

export { mailboxService, emailService, composeService, accountService, createProvider };
```

**cli.ts thin bootstrap:**
```typescript
#!/usr/bin/env bun
import { Command } from "commander";
import { initOAuthClient, initOutlookClient } from "./infrastructure/auth/index.js";
import { registerCommands } from "./commands/index.js";

const program = new Command();
program.name("mail-cli").description("CLI for reading and managing email").version("0.1.0");

// Register all commands
registerCommands(program);

// Global error handlers
process.on("uncaughtException", (err) => { printError(err); process.exit(1); });
process.on("unhandledRejection", (err) => { printError(err as Error); process.exit(1); });

// Initialize auth and parse
Promise.all([initOAuthClient(), initOutlookClient()])
  .then(() => { program.parse(); })
  .catch((err) => { printError(err); process.exit(1); });
```

**CRITICAL constraint:** The mark, move, delete commands call provider.mark(), provider.move(), provider.delete() directly since those methods are not yet in services. This is acceptable for Phase 7 - services will be expanded in later phases.
