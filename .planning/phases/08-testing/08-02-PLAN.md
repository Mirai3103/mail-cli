Commands use this pattern:
```typescript
export function registerListCommand(program: Command) {
  program
    .command("list")
    .description("...")
    .option("...")
    .action(async (options) => {
      try {
        const provider = await createProvider(options.account);
        const mailboxService = createMailboxService(provider);
        const result = await mailboxService.list(options.folder, limit);
        console.log(JSON.stringify(result));
      } catch (err) {
        const { printError } = await import("../utils/errors.js");
        printError(err as Error);
        process.exit(1);
      }
    });
}
```

Each command:
- Takes a Command from commander as input
- Registers itself on that command
- Has an async action that calls createProvider, creates service, calls service method
- Outputs JSON via console.log
- Catches errors, prints them, exits with code 1
